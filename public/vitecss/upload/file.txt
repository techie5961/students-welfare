Route::post('/test-aspfiy-webhook', function(Request $request) {
    // Get the secret key from env
    $secret_key = env('ASPFIY_SECRET_KEY');
    
    // Calculate expected signature
    $expected_signature = md5($secret_key);
    
    // Prepare test data - modify these values as needed
    $testData = [
        'event' => 'PAYMENT_NOTIFICATION',
        'data' => [
            'account_number' => $request->input('account_number', '1234567890'), // Use your test account number
            'amount' => $request->input('amount', 1000),
            // Add other fields as needed
        ]
    ];
    
    // Forward the request to your webhook endpoint
    $response = Http::withHeaders([
        'x-wiaxy-signature' => $expected_signature,
        'Content-Type' => 'application/json',
    ])->post('http://localhost:8000/aspfiy-webhook', $testData);
    
    return response()->json([
        'message' => 'Test request sent',
        'signature_used' => $expected_signature,
        'webhook_response' => $response->body(),
        'status_code' => $response->status()
    ]);
});
